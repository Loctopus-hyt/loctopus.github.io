【题面】
洛谷P1486 【NOI2004】郁闷的出纳员
【题解】
此题把我搞郁闷了好几天，是我太菜了QwQ
操作  III  : 正常插入。
操作 FFF : 正常查找。注意为第 kkk多！
操作 AAA : 把每个元素的值都加 kkk。 第一反应就是对全体元素打一个加法标记dif。
    可后来的元素怎么办呢？ 第一思路是每个元素的标记都单独区分，但这样
    显然是不行的，那么有没有一种办法使原先的加分标记对后来元素也有效
    呢，于是就有了这样一个等式 xxx1+++ difdifdif === xxx0
    所以稍稍改变插入操作，插入 x−difx-difx−dif 即可解决问题。
    接下来 AAA 操作就变成 dif+kdif+kdif+k？
    别忘了还有 minminmin , 又是个麻烦
    其实有了上面的分析，很容易得出 minminmin 的解决办法
    插入条件：min≤min≤min≤ xxx0 等价于 min−dif≤min - dif ≤min−dif≤ xxx0 −dif=x- dif = x−dif=x 1
    故完整 AAA 操作为dif+kdif + kdif+k 、min−kmin - kmin−k
操作 SSS : 把每个元素的值都减 kkk。同上可得 SSS 操作可转化为dif−kdif - kdif−k 、min+kmin + kmin+k
    那要走的元素呢？一下提供两种方法。
    一、暴力删除（此题可过）
     循环删除即可。如下

    while(pre(R,minn)!=-inf)
    {
        del(R,pre(R,minn));
        leave++;   //记录离开人数
	}

#include<cstdio>
#include<iostream>
#include<stdlib.h>
#define N 100500
#define inf 2000000005
using namespace std;
int n,x,leave,minn,dif,tot;
char opt;
int sum=0,R=0;
int size[N],v[N],num[N],rd[N],son[N][2];
void pushup(int p)
{
	size[p]=size[son[p][0]]+size[son[p][1]]+num[p];
}
void rotate(int &p,int d)
{
	int k=son[p][d^1];
	son[p][d^1]=son[k][d];
	son[k][d]=p;
	pushup(p);
	pushup(k);
	p=k;
}
void ins(int &p,int x)
{
	if(!p)
	{
		p=++sum;
		size[p]=num[p]=1;
		v[p]=x;
		rd[p]=rand();
		return; 
	}
	if(v[p]==x)
	{
		num[p]++;
		size[p]++;
		return;
	}
	int d=(x>v[p]);
	ins(son[p][d],x);
	if(rd[p]<rd[son[p][d]]) rotate(p,d^1);
	pushup(p);
}
void del(int &p,int x)
{
	if(!p) return;
	if(x<v[p]) del(son[p][0],x);
	else if(x>v[p]) del(son[p][1],x);
	else
	{
		if(!son[p][0]&&!son[p][1])
		{
			num[p]--; size[p]--;
			if(num[p]==0) p=0;
		}
		else if(son[p][0]&&!son[p][1])
		{
			rotate(p,1);
			del(son[p][1],x);
		}
		else if(!son[p][0]&&son[p][1])
		{
			rotate(p,0);
			del(son[p][0],x);
		}
		else if(son[p][0]&&son[p][1])
		{
			int d=(rd[son[p][0]]>rd[son[p][1]]);
			rotate(p,d);
			del(son[p][d],x);
		}
	}
	pushup(p);
}
int find(int p,int x)
{
	if(!p) return 0;
	if(size[son[p][0]]>=x) return find(son[p][0],x);
	else if(size[son[p][0]]+num[p]<x)
		return find(son[p][1],x-num[p]-size[son[p][0]]);
	else return v[p];
}
int pre(int p,int x)
{
	if(!p) return -inf;
	if(v[p]>=x) return pre(son[p][0],x);
	else return max(v[p],pre(son[p][1],x));
}
int main()
{
    scanf("%d%d",&n,&minn);
    for (int i=1;i<=n;i++)
    {
    	scanf("%s%d",&opt,&x);
        if (opt=='I'){if(x-dif>=minn){ins(R,x-dif);tot++;}}
        else if (opt=='A') minn-=x,dif+=x;
        else if (opt=='S')
        {
        	minn+=x,dif-=x;
        	while(pre(R,minn)!=-inf)
        	{
        		del(R,pre(R,minn));
        		leave++;
			}
		}
        else if (opt=='F') printf("%d\n",(tot-leave)<x?-1:find(R,tot-leave-x+1)+dif);
    }
    printf("%d",leave);
    return 0;
}
