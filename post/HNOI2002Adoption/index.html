【HNOI2002】营业额统计

2020-03-26 4 min read # 题解

题面：
洛谷P2234 【HNOI2002】营业额统计
题解：审题后可知该题主要是查找在已有数中一个数xxx的前驱（此题定义为权值xxx小于
   等于的最大节点）与后继（此题定义为权值xxx大于等于的最小节点），然后选择
   一个更接近xxx的数。
   套用Treap模板即可，涉及插入，寻找前驱、后继的操作。
注意点：有负数存在
完整代码：

#include<cstdio>
#include<iostream>
#include<stdlib.h>
#include<cmath> 
#define N 40000
#define inf 2000000005
using namespace std;
int n,small,big,ans;
int sum=0,R=0;
int v[N],size[N],rd[N],num[N],son[N][2];
inline int read()
{
	int num=0,flag=1;
	char c=getchar();
	while(c>'9'||c<'0'){if(c=='-')flag=-1;c=getchar();}
	while(c>='0'&&c<='9'){num=num*10+c-'0';c=getchar();}
	return num*flag;
}
void pushup(int p)
{
	size[p]=size[son[p][0]]+size[son[p][1]]+num[p];
}
void rotate(int &p,int d)
{
	int k=son[p][d^1];
	son[p][d^1]=son[k][d];
	son[k][d]=p;
	pushup(p);
	pushup(k);
	p=k;
}
void ins(int &p,int x)
{
	if(!p)
	{
		p=++sum;
		size[p]=num[p]=1;
		v[p]=x;
		rd[p]=rand();
		return;
	}
	if(v[p]==x)
	{
		size[p]++;
		num[p]++;
		return;
	}
	int d=(x>v[p]);
	ins(son[p][d],x);
	if(rd[p]<rd[son[p][d]]) rotate(p,d^1);
	pushup(p);
}
void del(int &p,int x)
{
	if(!p) return;
	if(x>v[p]) del(son[p][0],x);
	else if(x<v[p]) del(son[p][1],x);
	else
	{
		if(!son[p][0]&&!son[p][1])
		{
			size[p]--;num[p]--;
			if(size[p]==0)p=0;
		}
		else if(son[p][0]&&!son[p][1])
		{
			rotate(p,1);
			del(son[p][1],x);
		}
		else if(!son[p][0]&&son[p][1])
		{
			rotate(p,0);
			del(son[p][0],x);
		}
		else if(son[p][0]&&son[p][1])
		{
			int d=(rd[son[p][0]]>rd[son[p][1]]);
			rotate(p,d);
			del(son[p][d],x);
		}
	}
	pushup(p);
}
int pre(int p,int x)
{
	if(!p) return -inf;
	if(v[p]==x) return v[p];
	if(v[p]>=x) return pre(son[p][0],x);
	else return max(v[p],pre(son[p][1],x));
}
int suc(int p,int x)
{
	if(!p) return inf;
	if(v[p]==x) return v[p];
	if(v[p]<=x) return suc(son[p][1],x);
	else return min(v[p],suc(son[p][0],x));
}
int select(int x)
{
	int k;
	small=pre(R,x);
	big=suc(R,x);
	k=abs(x-small)<=abs(x-big)?small:big;
	if(small==-inf&&big==inf) k=0;
	ans=ans+abs(x-k);
	ins(R,x);
	return 0;
}
int main()
{
	n=read();
	for(int i=1;i<=n;i++)select(read());
	printf("%d",ans);
	return 0;
}
