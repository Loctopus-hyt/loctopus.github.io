【问题描述】
 一年一度的军训又要开始了。军训时，教官要求所有人按身高从小到大列队，
 现给定 nnn个人的一个队列，mmm次询问，每次询问第 iii个人到第 jjj个人中身高一样
 的人最多有多少个？
【输入数据】
 第一行两个整数 nnn 和 mmm(1≤n,m≤100000)(1 ≤ n, m ≤ 100000)(1≤n,m≤100000)
 第二行 nnn个整数 分别代表 nnn个人的身高（身高为整数精确到毫米），
 保证从低到高排列（有些人身高一样）
 接下来 mmm行，每行两个整数 iii 和 jjj，代表询问区间的起点和终点
【输出数据】
 mmm行，每行一个整数，代表区间 [i,j][i,j][i,j]中身高一样的人最多有多少个
【输入样例】
 10 3
 1600 1600 1601 1601 1601 1601 1603 1605 1605 1605
 2 3
 1 10
 5 10
【输出样例】
 1

【题解】

一、正解
  以样例为例 将数列转化为 (怎么转化看下面感受一下)
   111 222 111 222 333 444 111 111 222 333
  查询变成：1.查询 suc[i]suc[i]suc[i](第一个大于 a[i]a[i]a[i]的数的位置)
       2.查询 [suc[i]+1,j][suc[i]+1,j][suc[i]+1,j] 的RMQ
       3.比较后做出选择
       注：有些小细节
  于是乎，要预处理出后继位置和RMQ。
  完整代码：

#include<bits/stdc++.h> 
#define N 100500
#define M 30
using namespace std;
int n,m,l,r,a[N],num[N],suc[N],fmaxn[N][M];
int init()
{
	int i,j;
	for(i=1;i<=n;i++) fmaxn[i][0]=num[i];
	for(j=1;(1<<j)<=n;j++)
		for(i=1;i+(1<<j)-1<=n;i++)
			fmaxn[i][j]=max(fmaxn[i][j-1],fmaxn[i+(1<<(j-1))][j-1]);
	return 0;
}
int findsuc()
{
    suc[n]=n;
    for(int i=n-1;i>=1;i--)
        if(a[i]==a[i+1])suc[i]=suc[i+1];
    	else suc[i]=i;
    return 0;
}
int solve(int l,int r)
{
	int k=0,t=suc[l],ex;
	if(t>=r) return r-l+1;
	ex=t-l+1; l=t+1;
	while((1<<(k+1))<=r-l+1) k++;
	return max(ex,max(fmaxn[l][k],fmaxn[r-(1<<k)+1][k]));
}
int main()
{
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		if(a[i]==a[i-1]) num[i]=num[i-1]+1;
		else num[i]=1;
	}
	init(); findsuc();
	for(int i=1;i<=m;i++)
	{
		scanf("%d%d",&l,&r);
		printf("%d\n",solve(l, r));
	}
	return 0;
}

#include<bits/stdc++.h> 
#define N 100500
#define M 30
using namespace std;
int n,m,l,r,a[N],t[N],s[N],suc[N],pre[N],fmaxn[N][M];
int init()
{
	for(int i=1;i<=n;i++) t[i]=a[i];
	sort(t+1,t+n+1);
	for(int i=1;i<=n;i++)
		a[i]=lower_bound(t+1,t+n+1,a[i])-t;
	return 0;
}
int findsuc()
{
	memset(s,0,sizeof s);
	int i,top=0;
    for(i=n;i>=1;i--)
    {
        while(top&&a[s[top]]<=a[i])top--;
        suc[i]=s[top]; s[++top]=i;
    }
    for(i=n;i>=1;i--) if(suc[i]==0) suc[i]=n+1; else break;
    return 0;
}
int findpre()
{
	memset(s,0,sizeof s);
	int i,top=0;
    for(i=1;i<=n;i++)
    {
        while(top&&a[s[top]]>=a[i])top--;
        pre[i]=s[top]; s[++top]=i;
    }
    return 0;
}
int dp()
{
	int i,j,num;
	for(i=1;i<=n;i++) fmaxn[i][0]=1;
	for(j=1;(1<<j)<=n;j++)
	{
		for(i=1;i+(1<<j)-1<=n;i++)
		{
			fmaxn[i][j]=max(fmaxn[i][j-1],fmaxn[i+(1<<(j-1))][j-1]);
			if(a[i+(1<<(j-1))-1]!=a[i+(1<<(j-1))]) continue;
			num=a[i+(1<<(j-1))];
			fmaxn[i][j]=max(fmaxn[i][j],(min(suc[num]-1,i+(1<<j)-1)-max(pre[num]+1,i)+1));
		}
	}
	return 0;
}
int solve(int l,int r)
{
	int k=0,ans=0,num;
	while((1<<(k+1))<=r-l+1) k++;
	ans=max(fmaxn[l][k],fmaxn[r-(1<<k)+1][k]);
	if(a[l+(1<<(k))-1]!=a[l+(1<<(k))]) return ans;
	num=a[l+(1<<(k))];
	ans=max(ans,(min(suc[num]-1,r)-max(pre[num]+1,l)+1));
	return ans;
}
int main()
{

一、树状数组模板
  推荐：1.同机房小君同学的blog 树状数组
     2.另一篇 树状数组
  主要思想：差分
  对于一个数组a[i]a[i]a[i]，有差分序列b[i]b[i]b[i]:

b[i]={b[1]=a[1]i=1b[i]=a[i]−a[i−1]i⩾1b[i] = \begin{cases} b[1]=a[1] & i=1 \\ b[i]=a[i]-a[i-1] & i \geqslant 1 \end{cases} b[i]={b[1]=a[1]b[i]=a[i]−a[i−1]​i=1i⩾1​

  比如A={a1,a2,a3,a4,a5};A=\{a1,a2,a3,a4,a5\};A={a1,a2,a3,a4,a5};
  则有B={a1,a2−a1,a3−a2,a4−a3,a5−a4};B=\{a1,a2-a1,a3-a2,a4-a3,a5-a4\};B={a1,a2−a1,a3−a2,a4−a3,a5−a4};
  差分和前缀互为逆运算
  a[i]a[i]a[i]： 2 4 5 3 4 7 6 3 1
  b[i]b[i]b[i]： 2 2 1   -2 1 3   -1   -3   -2
  易得：将 a[ia[ia[i~j]j]j]都+x+x+x 等价于在b[i]b[i]b[i]处+x+x+x，在b[j+1]b[j+1]b[j+1]处−x-x−x
  差分和前缀和可以算是处理序列的小技巧，有时候可以用得上。
  比如：留坑待填...

#include<bits/stdc++.h> 
#define N 100500
using namespace std;
int n,l,r,c[N]; 
inline int read()
{
	char c=getchar();
	int num=0,flag=1;;
	while(c<'0'||c>'9'){if(c=='-')flag=-1;c=getchar();}
	while(c>='0'&&c<='9'){num=num*10+c-'0';c=getchar();}
	return num*flag;
}
int lowbit(int n)
{
	return n&(-n);
} 
int update(int pos,int d)
{
	while(pos<=n)
	{
		c[pos]+=d;
		pos+=lowbit(pos);
	}
	return 0;
}
int sum(int pos)
{
	int ans=0;
	while(pos)
	{
		ans+=c[pos];
		pos-=lowbit(pos);
	}
	return ans;
}
int main()
{
	n=read();
	for(int i=1;i<=n;i++)
	{
		l=read(); r=read();
		update(l,1);
		update(r+1,-1);
	}
	for(int i=1;i<=n;i++)printf("%d ",sum(i));
}

  由于样例不具有代表性，手画人造数据，模拟数据是OIer解题时的一种重要方式

//f[i][j]初始值为-1    以i为根选了j个黑色节点
int dp(int x,int fa)
{
	f[x][0]=f[x][1]=0;
    //tot表示当前所枚举的子树(根为下面的son)的总结点数
    //sum表示该树(根为x)的总结点数
	int sum=1,tot=0;
	ll ans=0;
	for(int i=first[x];i;i=next[i])
	{
		int son=go[i];
		if(son==fa) continue;
		tot=dp(son,x); sum+=tot;
        //  外重循环 倒序！倒序！倒序！
		for(int j=min(m,sum);j>=0;j--)  //当前树选j个黑色节点 
			for(int k=0;k<=min(j,tot);k++)  //子树选k个黑色节点
			{
				if(f[x][j-k]==-1)continue;
                //上文提到的花费，也可称之为贡献
				ans=k*(m-k)*w[i]+(tot-k)*(n-m-tot+k)*w[i];
				f[x][j]=max(f[x][j],f[x][j-k]+f[son][k]+ans);
			}
	}
	return sum;
}
